Running Sum of 1d Array [1480]

    
class Solution(object):
    def runningSum(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        t = 0
        li = []
        for i in range(0, len(nums)):
            t = t+nums[i]
            li.append(t)
        return li

x = Solution()
print(x.runningSum([3,1,2,10,1]))
